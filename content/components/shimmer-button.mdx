## Prerequisites

```bash
npm install -D tailwindcss clsx tailwind-merge
```

Copy and paste the following code into your project.

```typescript
// lib/utils.ts
import clsx, { ClassValue } from "clsx";
import { twMerge } from "tailwind-merge";

export function cn(...inputs: ClassValue[]) {
  return twMerge(clsx(inputs));
}
```

```js
// tailwind.config.js
module.exports = {
  theme: {
    extend: {
      animation: {
        flip: "flip calc(var(--spark) * 2) infinite steps(2, end)",
        rotate: "rotate var(--spark) linear infinite both;",
      },
      keyframes: {
        flip: {
          to: { rotate: "360deg" },
        },
        rotate: {
          to: {
            transform: "rotate(90deg)",
          },
        },
      },
    },
  },
};
```

## Props

| Prop Name     | Type            | Description                         | Default Value                                                                   |
| ------------- | --------------- | ----------------------------------- | ------------------------------------------------------------------------------- |
| shadow        | string          | The shadow of the button            |                                                                                 |
| shadowEnabled | boolean         | Whether the shadow is enabled       | true                                                                            |
| shimmerColor  | string          | The color of the shimmer            | #ffffff                                                                         |
| shimmerSize   | string          | The size of the shimmer             | 0.1em                                                                           |
| borderRadius  | string          | The border radius of the button     | 100px                                                                           |
| sparkDuration | string          | The duration of the spark animation | 1.8s                                                                            |
| background    | string          | The background of the button        | radial-gradient(ellipse 80% 50% at 50% 120%,rgba(62, 61, 117),rgba(13, 13, 33)) |
| className     | string          | The class name of the button        | undefined                                                                       |
| children      | React.ReactNode | The children of the button          | undefined                                                                       |
